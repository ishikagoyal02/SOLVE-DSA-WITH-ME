<h1><a href="https://leetcode.com/problems/design-hashmap/description/">Design HashMap</a></h1>

```cpp

 int mp[1000001];
    MyHashMap() {
        for(int i=0;i<1000001;i++)
            mp[i]=-1;
    }
    
    void put(int key, int value) {
        mp[key]=value;
    }
    
    int get(int key) {
        return mp[key];
    }
    
    void remove(int key) {
        mp[key]=-1;
    }
```
