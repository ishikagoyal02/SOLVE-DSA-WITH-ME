    bool isSubsequence(string s, string t) {
        
        int i=0;
        int j=0;
        while(i<s.size() && j<t.size())
        {
            if(s[i]==t[j])
            {
                i++;
                j++;
            }
            else
                j++;
        }
        //cout<<endl<<i;
        if(i!=s.size())
            return false;
        return true;    
    }
