<h3><a href="https://leetcode.com/problems/kth-largest-element-in-an-array/"> Kth Largest Element in an Array</a></h3>

```cpp
 int findKthLargest(vector<int>& nums, int k) {
        
        priority_queue<int> q;

        for(int i=0;i<nums.size();i++)
            q.push(nums[i]);

        for(int i=0;i<k-1;i++)
            q.pop();

        return q.top();        
    }
